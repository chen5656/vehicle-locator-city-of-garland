/****** Script for SelectTopNRows command from SSMS  ******/
SELECT TOP 1000 [Id]
      ,[Vehicle]
      ,[LoginName]
      ,[CrewId]
      ,[CrewNumber]
      ,[DeptId]
      ,[Speed]
      ,[Heading]
      ,[LocationDate]
      ,[Location]
  FROM [AVL].[dbo].[AVL_History]
  WHERE [AVL].[dbo].[AVL_History].[LocationDate] BETWEEN '11/11/2015 00:00:00.00' AND '11/11/2015 23:59:59.999'
